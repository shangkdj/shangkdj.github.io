def nsfile(counts,ddir,size):
    '''The number of new expected documents'''

#生成文件
    for i in range(1,counts+1):

        localTime = time.strftime("%Y%m%d%H%M%S",time.localtime())

        #print localtime
        filename = str(ddir)+"/"+localTime+"_"+str(i)+".txt"

        #a:以追加模式打开（必要时可以创建）append;b:表示二进制
        f = open(filename,'ab')

        f.write(os.urandom(size))
        f.close()

        #输出第几个文件和对应的文件名称
        print "file"+" "+str(i)+":"+str(localTime)+".txt"
    print "ALL MAKE OVER"

#r需要删除的目录
def rm_dir(rdir):
    '''rm file'''
    if os.path.exists(rdir):

        shutil.rmtree(rdir)
        print "remove --%s ok"%rdir
    else:
        print "%s is not exists"%rdir


#cat
def cat_file(catdir):
    '''cat file number'''

    filecount=0

    for root,dir,files in os.walk(catdir):
        filecount+=len(files)

    return filecount

def get_fsize(path):
    count = 0
    for dirname in os.listdir(path):
        tmp_dir = os.path.join(path,dirname)
        if (os.path.isdir(tmp_dir)):
            for r,ds,files in os.walk(tmp_dir):
                for f1 in files:
                    size = os.path.getsize(os.path.join(r,f1)) 
                    count += size

    return count
